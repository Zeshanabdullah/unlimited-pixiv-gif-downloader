This example demonstrates how to download a GIF using the requests library in Python.

import requests

def download_gif(gif_url, save_path):
    try:
        response = requests.get(gif_url)
        response.raise_for_status()  # Raise an error for bad responses
        with open(save_path, 'wb') as f:
            f.write(response.content)
        print(f"GIF downloaded successfully: {save_path}")
    except requests.exceptions.RequestException as e:
        print(f"Error downloading GIF: {e}")

### PHP Example Using cURL
This example shows how to use cURL in PHP to download a GIF from the given URL.

php
<?php

function downloadGif($gifUrl, $savePath) {
    $ch = curl_init($gifUrl);
    $fp = fopen($savePath, 'wb');

    curl_setopt($ch, CURLOPT_FILE, $fp);
    curl_setopt($ch, CURLOPT_FOLLOWLOCATION, true);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);

    if(curl_exec($ch) === false) {
        echo 'Curl error: ' . curl_error($ch);
    } else {
        echo "GIF downloaded successfully: {$savePath}";
    }

    curl_close($ch);
    fclose($fp);
}

// Example usage
$gifUrl = 'https://hdstockimages.com/pixiv-gif-downloader/sample.gif';
